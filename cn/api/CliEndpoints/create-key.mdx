---
title: '创建新 KEY'
description: '创建一个新的 API KEY'
api: 'POST /api/token/'
---

<ParamField body="name" type="string" required>
  新 KEY 名称
</ParamField>

<ParamField body="expired_time" type="number" default={-1}>
  过期时间（-1 表示永不过期；也可传秒数）
</ParamField>

<ParamField body="remain_quota" type="number" default={500000}>
  初始额度（示例：500000）
</ParamField>

<ParamField body="unlimited_quota" type="boolean" default={false}>
  是否无限额度
</ParamField>

<ParamField body="subnet" type="string" default="">
  子网限制（示例为空字符串）
</ParamField>

<ResponseField name="success" type="boolean" />
<ResponseField name="message" type="string" />
<ResponseField name="data" type="object">
  通常包含新 key 与 id
</ResponseField>
